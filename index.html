Method Testing

        
class Klazz
  def method
    do_something
  end
end

class KlazzTest
  def setup
    @klazz = Klazz.new
  end
  def test_method
    result = @klazz.do_something
    assert_equal("expected", result)
  end
end

The Context Problem

      
class KlazzTest
  def setup
    @klazz = Klazz.new
  end
  def test_method
    result = @klazz.do_something
    assert_equal(:expected, result)
  end
  def test_another_context
    klazz = Klazz.new(:constructor_args)
    assert_equal(:expected_2, klazz.do_something)
  end
end
